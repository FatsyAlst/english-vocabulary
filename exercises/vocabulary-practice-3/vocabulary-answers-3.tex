\section{Exercise 10: Trap Sentences (12 points)}

\begin{answerbox}[Answer Key]
\begin{enumerate}
    \item Error: \textbf{ceased} → Correction: \textbf{seized}
    
    \textit{Explanation: "Ceased" means "stopped," while "seized" means "took by force/confiscated." The police seized (not ceased) the weapons.}
    
    \item Error: \textbf{brazing} → Correction: \textbf{brazen}
    
    \textit{Explanation: "Brazing" is a metalworking term (welding), while "brazen" means shamelessly bold. "Brazen disregard" is the correct collocation.}
    
    \item Error: \textbf{mischievious} → Correction: \textbf{mischievous}
    
    \textit{Explanation: Common spelling error. The word has only 3 syllables (MIS-chiv-ous), not 4. No extra "i" before "-ous."}
    
    \item Error: \textbf{fallen on the wayside} → Correction: \textbf{fallen by the wayside}
    
    \textit{Explanation: The correct idiom is "fall BY the wayside," not "on." The preposition matters!}
    
    \item Error: \textbf{therefore} → Correction: \textbf{thereby}
    
    \textit{Explanation: "Thereby" means "by that means/action" and connects to the preceding gerund. "Therefore" means "for that reason" and is used differently.}
    
    \item Error: \textbf{scattered together} → Correction: \textbf{scattered} (or \textbf{gathered together})
    
    \textit{Explanation: "Scatter" means to separate and go in different directions — the opposite of "together." Either remove "together" or change to "gathered."}
\end{enumerate}
\end{answerbox}

\textbf{Scoring:} 1 point for identifying error, 1 point for correct correction (12 total)

\section{Exercise 13: Common Confusion Pairs (8 points)}

\begin{answerbox}[Answer Key]
\begin{enumerate}
    \item \textbf{b) seized} — took by force (ceased = stopped)
    \item \textbf{b) ensued} — followed as a result (ensured = guaranteed)
    \item \textbf{b) dicey} — risky (dicy is not a word)
    \item \textbf{a) brazen} — shameless (brazing = welding metal)
    \item \textbf{a) canopy} — tree cover (canapé = small appetizer)
    \item \textbf{a) savory} — salty/spicy, not sweet (savvy = knowledgeable)
    \item \textbf{b) thereby} — by that means (therefore = for that reason)
    \item \textbf{a) mischievous} — correct spelling (mischievious = common misspelling)
\end{enumerate}
\end{answerbox}

\begin{tipbox}
\textbf{Key Distinctions:}
\begin{itemize}
    \item \textbf{ceased/seized}: Ceased = stopped doing; Seized = grabbed/confiscated
    \item \textbf{ensured/ensued}: Ensured = guaranteed; Ensued = followed as result
    \item \textbf{thereby/therefore}: Thereby = by that action; Therefore = for that reason
    \item \textbf{savory/savvy}: Savory = food taste (salty); Savvy = knowledge/skill
\end{itemize}
\end{tipbox}
